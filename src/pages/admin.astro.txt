    <script>
        const PASSWORD = "Harsh#2404";
        
        const loginForm = document.getElementById('loginForm');
        const publishForm = document.getElementById('publishForm');
        const passwordInput = document.getElementById('password');
        const loginBtn = document.getElementById('loginBtn');
        const loginError = document.getElementById('loginError');
        const generateBtn = document.getElementById('generateBtn');
        const logoutBtn = document.getElementById('logoutBtn');
        const publishStatus = document.getElementById('publishStatus');
        const markdownContainer = document.getElementById('markdownContainer');
        const markdownOutput = document.getElementById('markdownOutput');
        const copyBtn = document.getElementById('copyBtn');

        loginBtn.addEventListener('click', () => {
            if (passwordInput.value === PASSWORD) {
                loginForm.classList.add('hidden');
                publishForm.classList.remove('hidden');
                loginError.textContent = '';
            } else {
                loginError.textContent = '❌ incorrect password';
                passwordInput.value = '';
            }
        });

        passwordInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') loginBtn.click();
        });

        logoutBtn.addEventListener('click', () => {
            publishForm.classList.add('hidden');
            loginForm.classList.remove('hidden');
            loginError.textContent = '';
            passwordInput.value = '';
            document.getElementById('title').value = '';
            document.getElementById('description').value = '';
            document.getElementById('content').value = '';
            document.getElementById('tags').value = '';
            publishStatus.textContent = '';
            markdownContainer.classList.add('hidden');
        });

        generateBtn.addEventListener('click', () => {
            const title = document.getElementById('title').value.trim();
            const description = document.getElementById('description').value.trim();
            const content = document.getElementById('content').value.trim();
            const tags = document.getElementById('tags').value.split(',').map(t => t.trim()).filter(t => t);

            if (!title || !content) {
                publishStatus.textContent = '❌ title and content required';
                publishStatus.classList.remove('hidden');
                return;
            }

            const slug = title
                .toLowerCase()
                .replace(/[^\w\s-]/g, '')
                .replace(/\s+/g, '-')
                .replace(/-+/g, '-')
                .slice(0, 50);

            const date = new Date().toISOString().split('T')[0];

            const markdown = `---
title: "${title.replace(/"/g, '\\"')}"
description: "${description.replace(/"/g, '\\"')}"
pubDate: "${date}"
author: "Harsh"
tags: [${tags.map(t => `"${t}"`).join(', ')}]
layout: ../../layouts/BlogPost.astro
---

${content}`;

            markdownOutput.textContent = markdown;
            markdownContainer.classList.remove('hidden');
            publishStatus.textContent = '✅ markdown generated! copy it below.';
            publishStatus.classList.remove('hidden');

            copyBtn.dataset.markdown = markdown;
        });

        copyBtn.addEventListener('click', () => {
            const markdown = copyBtn.dataset.markdown;
            navigator.clipboard.writeText(markdown).then(() => {
                publishStatus.textContent = '✅ copied to clipboard! now create src/pages/blog/' + document.getElementById('title').value.toLowerCase().replace(/\s+/g, '-') + '.md and paste it.';
                publishStatus.classList.remove('hidden');
            });
        });
    </script>
